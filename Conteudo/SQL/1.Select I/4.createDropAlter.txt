CREATE TABLE: Permite criar uma tabela do zero
DROP TABLE: Atua apenas na tabela que queremos excluir
ALTER TABLE: Permite alterar uma tabela podendo operar com 3 comandos:
  -> ADD: Adicionar coluna
  -> MODIFY: Modificar uma coluna
  -> DROP: Excluir uma coluna
